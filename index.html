// 1. 加入購物車邏輯 const addToCart = () => { if (!appUser || !pendingOrder) return; if (!orderConfig.sugar && !sugarLocked) { showNotify('請選擇甜度！', 'error'); setValidationError('sugar'); return; } if (!orderConfig.ice && !iceLocked) { showNotify('請選擇冰塊！', 'error'); setValidationError('ice'); return; } const finalPrice = calculateCurrentTotal(); const toppingNames = orderConfig.toppings.map(t => t.name); const cartItem = { cartId: Date.now() + Math.random(), // 購物車專用暫時 ID userName: appUser.name, storeKey: activeStoreKey, storeName: currentStore.name, itemName: pendingOrder.item.name, size: pendingOrder.size, price: finalPrice, sugar: String(orderConfig.sugar), ice: String(orderConfig.ice), toppings: toppingNames, timestamp: Date.now(), userId: user ? user.uid : 'anonymous', isPickedUp: false }; setCart(prev => [...prev, cartItem]); showNotify(`已加入購物車：${pendingOrder.item.name}`); setModalOpen(false); setPendingOrder(null); }; // 2. 購物車結帳邏輯 (一次送出多筆) const submitCart = async () => { if (cart.length === 0) return; setLoading(true); try { const batch = writeBatch(db); // 使用 Batch 一次打包送出 cart.forEach(item => { const { cartId, ...firebaseData } = item; // 移除暫時 ID const ref = doc(collection(db, 'orders')); batch.set(ref, firebaseData); }); await batch.commit(); showNotify(`成功送出 ${cart.length} 杯飲料！`); setCart([]); // 清空購物車 } catch (e) { showNotify('送出失敗', 'error'); } setLoading(false); };
